 21 /**
 22  * @param  {} size, tamaño del tablero
 23  * @description funcion utilizada para inicializar la matriz del tablero junto a sus
 24  * correspondientes atributos
 25  */
 26 function constructor(size) {
 27   tablero.size = size;
 28   for (let i = 0; i < size; i++) {
 29     tablero.fila[i] = []
 30     for (let j = 0; j < size; j++) {
 31       tablero.fila[i][j] = ".";
 32     }
 33 
 34     tablero.reinasEnFila[i] = false;
 35     tablero.reinasEnColumna[i] = false;
 36   }
 37 }

 39 /**
 40  * @description Funcion utilizada para imprimi el tablero junto a las reinas que ha sido colocadas
 41  */
 42 function imprimirTablero() {
 43   for (let i = 0; i < tablero.size; i++) {
 44     let cadena = "";
 45 
 46     for (let j = 0; j < tablero.fila[i].length; j++) {
 47       cadena += " " + tablero.fila[i][j];
 48     }
 49     console.log(cadena);
 50   }
 51   console.log();
 52 }

 54 /**
 55  * @param  {} columna, columna en la que queremos insertar la nueva reina.
 56  * @description, En esta funcion se ejecuta de manera recursiva, la llamada a insertarReina(),
 57  * donde se van insertando las reinas en aquellas posiciones validas. Empleando recursividad 
 58  * para generar todas las combinaciones posibles
 59  */
 60 function insertarReina(columna) {
 61   if (tablero.reinasAlmacenadas.length === tablero.size) {
 62     imprimirTablero();
 63     contador++;
 64   }
 65   else {
 66     for (let fila = 0; fila < tablero.fila.length; fila++) {
 67       if (comprobarFila(fila) && comprobarColumna(columna) && comprobarDiagonalNormal(fila, columna) && comprobarLinea(fila, columna)) {
 68         crearReina(fila, columna);
 69         insertarReina(columna + 1);
 70         eliminarReina();
 71       }
 72     }
 73   }
 74 }

 76 /**
 77  * @param  {} fila, coordenada en el eje X en la que se va a insertar la reina
 78  * @param  {} columna, coordenada en el eje Y en la que se va a insertar la reina
 79  * @description, En esta funcion, su principal finalidad, es la de crear un nuevo
 80  * objeto de tipo reina que insertaremos en las coordenadas que se le pasa como
 81  * argumentos, ya que han sido comprobadas previamente y son validas
 82  */
 83 function crearReina(fila, columna) {
 84   let nuevaReina = {fila: fila, columna: columna};
 85   nuevaReina.fila = fila;
 86   nuevaReina.columna = columna;
 87 
 88   tablero.fila[fila][columna] = "&";
 89   tablero.reinasEnFila[fila] = true;
 90   tablero.reinasEnColumna[columna] = true;
 91   tablero.reinasAlmacenadas.push(nuevaReina);
 92 }

 94 /**
 95  * @description, Necesitamos de esta implementacion debido a que cuando vuelva a atras
 96  * en la recursividad, el motivo de esa vuelta atras es porque si colocamos esa reina en esa
 97  * casilla, implica que ya no podemos colocar mas reinas, por lo que esa reina no va en esa
 98  * casilla, por lo que procedemos a eliminarla, para poder colocarla en otra que si nos de
 99  * una solucion correcta
100  */
101 function eliminarReina() {
102   let ejeX = tablero.reinasAlmacenadas[tablero.reinasAlmacenadas.length - 1].fila;
103   let ejeY = tablero.reinasAlmacenadas[tablero.reinasAlmacenadas.length - 1].columna;
104 
105   tablero.reinasEnFila[ejeX] = false;
106   tablero.reinasEnColumna[ejeY] = false;
107   tablero.fila[ejeX][ejeY] = ".";
108   tablero.reinasAlmacenadas.pop();
109     
110 }

112 /**
113  * @param  {} fila, coordenada en el eje X, de la reina que queremos introducir en la posicion dada
114  * @param  {} columna, coordenada en el eje Y, de la reina que queremos introducir en la posicion dada
115  * @description, Funcion utilizada para generar todas las combinaciones posibles de rectas que se pueden
116  * generar con las reinas ya introducidas en el tablero
117  */
118 function comprobarLinea(fila, columna) {
119   if (tablero.reinasAlmacenadas.length < 2) {
120     return true;
121   }
122   else {
123     for (let reinaPrimera = 0; reinaPrimera < tablero.reinasAlmacenadas.length; reinaPrimera++) {
124       for (let reinaSegunda = reinaPrimera + 1; reinaSegunda < tablero.reinasAlmacenadas.length; reinaSegunda++) {
125         if (isPointInLine(fila, columna, lineFromTo(tablero.reinasAlmacenadas[reinaPrimera],
126           tablero.reinasAlmacenadas[reinaSegunda]))) {
127             return false;
128         }
129       }
130     }
131   }
132 
133   return true;
134 }

136 /**
137  * @param  {} fila, coordena en el eje X, perteneciente a la reina que queremos colocar
138  * @param  {} columna, coordena en el eje Y, perteneciente a la reina que queremos colocar
139  * @param  {} linea, line formada por dos reinas dadas que utilizaremos para comprobar
140  * @description, Funcion usada para comprobar si un punto dado pertenece a la recta
141  * formada por dos reinas
142  */
143 function isPointInLine(fila, columna, linea) {
144   return (((linea.pendiente * fila) + linea.desplazamiento) - columna) === 0;
145 }
146 
147 /**
148  * @param  {} reinaPrimera, parametro uno, usado como primer punto que utilizaremos para
149  * trazar la recta entre los dos puntos
150  * @param  {} reinaSegunda, parametro dos, usado como segundo punto que utilizaremos para
151  * trazar la recta entre los dos puntos
152  * @description, Funcion utilizada para obtener los parametros que se utilizan para comprobar
153  * si un punto dado pertenece a la recta que forman dos reinas. Para obtener le recta usamos 
154  * las formula -> y = mx + n
155  *  · m = (y2 - y1) / (x2 - x1)
156  *  · n = 
157  */
158 function lineFromTo(reinaPrimera, reinaSegunda) {
159   let pendiente_ = ((reinaSegunda.columna - reinaPrimera.columna) / (reinaSegunda.fila - reinaPrimera.fila));
160   let desplazamiento_ = reinaPrimera.columna + (pendiente_ * (-reinaPrimera.fila));
161 
162   return {pendiente: pendiente_, desplazamiento: desplazamiento_};
163 }

165 /**
166  * @param  {} fila, comprobamos que la fila en la que queremos insertar la reina es valida
167  * y no hay otra reian en ese misma fila pero en otra columna que amenace a la reina

169  * insertada o no. Por ello simplemente comprobamos si en tablero.reinasEnFila[fila] hay un true
170  * @returns, En caso de encontrar que en esa fila, ya se encuentra una reina insertada, retornamos

172  */

175     return false;
176   }

178   return true;
179 }

197 /**
198  * @param  {} fila, coordenada respecto al eje X, que queremos comprobar si esta o no en la diagonal
199  * de alguna reina para que no la amenace
200  * @param  {} columna, coordenada respecto al eje X, que queremos comprobar si esta o no en la diagonal
201  * de alguna reina para que no la amenace
202  * @returns {boolean} Una vez comprobada la posicion, en caso de que pertenezca a una diagonal, se retornaria
203  * que esa posicion no es valida, por lo que se retorna false. En caso de si ser valida, se retorna true
204  */
205 function comprobarDiagonalNormal(fila, columna) {
206   let ejeX = fila;
207   let ejeY = columna;
208   const REINA = "&";
209 
210   // Recorremos la diagonal del primer cuadrante respecto a la posicion (todo positivo, 1º cuadrante)
211   while (ejeX >= 0 && ejeY < tablero.fila.length) {
212     if (tablero.fila[ejeX][ejeY] === REINA) {
213       return false;
214     }
215 
216     ejeX--;
217     ejeY++;
218   }
219 
220   // Recorremos la diagonal del segundo cuadrante respecto a la posicion (ejeX negativo y ejeY positivo, 2º cuadrante)
221   ejeX = fila;
222   ejeY = columna;
223   while (ejeX >= 0 && ejeY >= 0) {
224     if (tablero.fila[ejeX][ejeY] === REINA) {
225       return false;
226     }
227 
228     ejeX--;
229     ejeY--;
230   }
231 
232   // Recorremos la diagonal del tercer cuadrante respecto a la posicion (ejeX negativo y ejeY negativo, 3º cuadrante)
233   ejeX = fila;
234   ejeY = columna;
235   while (ejeX < tablero.fila.length && ejeY >= 0) {
236     if (tablero.fila[ejeX][ejeY] === REINA) {
237       return false;
238     }
239 
240     ejeX++;
241     ejeY--;
242   }
243 
244   // Recorremos la diagonal del cuarto cuadrante respecto a la posicion (ejeX positivo y ejeY positivo, 4º cuadrante)
245   ejeX = fila;
246   ejeY = columna;
247   while (ejeX < tablero.fila.length && ejeY < tablero.fila.length) {
248     if (tablero.fila[ejeX][ejeY] === REINA) {
249       return false;
250     }
251 
252     ejeX++;
253     ejeY++;
254   }
255 
256   return true;
257 }
